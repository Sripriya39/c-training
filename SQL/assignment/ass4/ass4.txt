declare @number int = 3; 
declare @factorial int = 1;

while @number > 1
begin
    set @factorial = @factorial * @number;
    set @number = @number - 1;
end

print 'The factorial of the given number is: ' + convert(varchar, @factorial);



create or alter procedure GenerateMultiplicationTable
    @Number int,
    @Limit int
as
begin
    declare @i int
    set @i = 1
    print 'Multiplication table for ' + CONVERT(VARCHAR, @Number) + ':'
    while @i <= @Limit
    begin
        print convert(VARCHAR, @Number) + ' * ' + CONVERT(VARCHAR, @i) + ' = ' + CONVERT(VARCHAR, @Number * @i)
        set @i = @i + 1
    end
end

exec GenerateMultiplicationTable 5, 10

create procedure sp_get_date
    @date DATE
as
begin
 
    DECLARE @holiday_name VARCHAR(50);

    SELECT @holiday_name = holiday_name
    FROM Holidays
    WHERE holiday_date = @date;

    IF @holiday_name IS NOT NULL
    BEGIN
        raiseerror ('Due to %s, you cannot manipulate data', 16, 1, @holiday_name);
    END
END
